let selectedEye = ''; let currentICD = ''; let currentDiagLabel = ''; // Pre-load logo let logoDataURL = null; const logoImg = new Image(); logoImg.crossOrigin = 'anonymous'; logoImg.src = 'MSHS-NYEE-Logo-300x12.png'; logoImg.onload = function() { const canvas = document.createElement('canvas'); canvas.width = logoImg.naturalWidth; canvas.height = logoImg.naturalHeight; canvas.getContext('2d').drawImage(logoImg, 0, 0); logoDataURL = canvas.toDataURL('image/png'); }; const lateralityEmbedded = ['wAMD','DME','BRVO','CRVO']; function onDiagCategoryChange() { const cat = document.getElementById('diagCategory').value; // Hide all sub-selectors and reset their values ['wAMD','DME','BRVO','CRVO'].forEach(k => { const el = document.getElementById('sub-'+k); if (el) { el.classList.remove('visible'); const sel = document.getElementById('sel-'+k); if (sel) sel.value = ''; } }); // Show the relevant sub-selector if (cat && document.getElementById('sub-'+cat)) { document.getElementById('sub-'+cat).classList.add('visible'); } // Show/hide eye buttons document.getElementById('eyeRow').style.display = lateralityEmbedded.includes(cat) ? 'none' : 'block'; // Reset eye selection selectedEye = ''; ['OD','OS','OU'].forEach(e => { const b = document.getElementById('btn-'+e); if (b) b.classList.remove('selected'); }); // Reset ICD state currentICD = ''; currentDiagLabel = ''; document.getElementById('icdPreview').textContent = ''; } function selectEye(eye) { selectedEye = eye; ['OD','OS','OU'].forEach(e => document.getElementById('btn-'+e).classList.remove('selected')); document.getElementById('btn-'+eye).classList.add('selected'); updateICD(); } const simpleDiagMap = { ME: { OD: { code:'H35.81', label:'Macular Edema, Unspecified, right eye' }, OS: { code:'H35.82', label:'Macular Edema, Unspecified, left eye' }, OU: { code:'H35.89', label:'Macular Edema, Unspecified, bilateral' } }, uveitis: { OD: { code:'H30.001', label:'Unspecified Focal Chorioretinal Inflammation, right eye' }, OS: { code:'H30.002', label:'Unspecified Focal Chorioretinal Inflammation, left eye' }, OU: { code:'H30.003', label:'Unspecified Focal Chorioretinal Inflammation, bilateral' } } }; function updateICD() { const cat = document.getElementById('diagCategory').value; const preview = document.getElementById('icdPreview'); currentICD = ''; currentDiagLabel = ''; preview.textContent = ''; if (!cat) return; if (lateralityEmbedded.includes(cat)) { // Read from the sub-dropdown for this category const selEl = document.getElementById('sel-' + cat); if (!selEl || !selEl.value) { preview.textContent = 'Select a specific code above.'; return; } const parts = selEl.value.split('|'); currentICD = parts[0].trim(); currentDiagLabel = parts[1] ? parts[1].trim() : parts[0].trim(); preview.textContent = 'ICD-10: ' + currentICD + ' — ' + currentDiagLabel; } else { // ME or uveitis — need eye button selection if (!selectedEye) { preview.textContent = 'Select affected eye to see ICD-10 code.'; return; } const entry = simpleDiagMap[cat] && simpleDiagMap[cat][selectedEye]; if (!entry) return; currentICD = entry.code; currentDiagLabel = entry.label; preview.textContent = 'ICD-10: ' + currentICD + ' — ' + currentDiagLabel; } } function updateCursivePreview() { document.getElementById('cursivePreview').textContent = document.getElementById('signerName').value.trim(); } function formatDate(s) { if (!s) return ''; const [y,m,d] = s.split('-'); return `${m}/${d}/${y}`; } function getTodayFormatted() { return new Date().toLocaleDateString('en-US',{year:'numeric',month:'long',day:'numeric'}); } function buildPDF(logo) { const name = document.getElementById('patientName').value.trim(); const dob = document.getElementById('dob').value; const requested = document.getElementById('requestedDrug').value; const signer = document.getElementById('signerName').value.trim(); const failed = Array.from(document.querySelectorAll('.checkbox-grid input:checked')).map(c=>c.value); const failedList = failed.length===1 ? failed[0] : failed.slice(0,-1).join(', ') + ', and ' + failed[failed.length-1]; // Determine laterality label for header const cat = document.getElementById('diagCategory').value; let eyeLabelCap = ''; if (lateralityEmbedded.includes(cat)) { const dl = currentDiagLabel.toLowerCase(); if (dl.includes('bilateral')) eyeLabelCap = 'Both Eyes (OU)'; else if (dl.includes('right')) eyeLabelCap = 'Right Eye (OD)'; else if (dl.includes('left')) eyeLabelCap = 'Left Eye (OS)'; else eyeLabelCap = 'Unspecified Eye'; } else { eyeLabelCap = selectedEye==='OD' ? 'Right Eye (OD)' : selectedEye==='OS' ? 'Left Eye (OS)' : 'Both Eyes (OU)'; } const eyeInBody = eyeLabelCap==='Both Eyes (OU)' ? 'both eyes (OU)' : eyeLabelCap==='Right Eye (OD)' ? 'the right eye (OD)' : 'the left eye (OS)'; const {jsPDF} = window.jspdf; const doc = new jsPDF({unit:'pt', format:'letter'}); const mL=72, mR=72, pW=612, tW=pW-mL-mR; let y = 52; const sf = (style, size, color=[0,0,0]) => { doc.setFont('helvetica', style); doc.setFontSize(size); doc.setTextColor(...color); }; const wrap = (text, yy, lh=15.5) => { const lines = doc.splitTextToSize(text, tW); lines.forEach((l,i) => doc.text(l, mL, yy + i*lh)); return yy + lines.length * lh; }; // Logo if (logo) { doc.addImage(logo, 'PNG', mL, y, 240, 96); y += 110; } else { sf('bold', 13, [26,46,74]); doc.text('New York Eye and Ear Infirmary of Mount Sinai', mL, y+30); y += 50; } doc.setDrawColor(200,210,220); doc.setLineWidth(0.75); doc.line(mL,y,pW-mR,y); y+=14; sf('bold',11,[26,46,74]); doc.text('Department of Ophthalmology — Vitreoretinal Surgery & Medical Retina', mL, y); y+=22; sf('normal',10,[100,100,100]); doc.text(getTodayFormatted(), mL, y); y+=26; // RE block sf('bold',10.5,[0,0,0]); doc.text(`RE: Medical Necessity — ${requested}`, mL, y); y+=15; sf('normal',10.5); doc.text(`Patient: ${name} | DOB: ${formatDate(dob)} | ${eyeLabelCap}`, mL, y); y+=15; // Wrap diagnosis line in case it's long y = wrap(`Diagnosis: ${currentDiagLabel} | ICD-10: ${currentICD}`, y); y+=14; // Body sf('normal',10.5); y = wrap('To Whom It May Concern:', y); y+=14; y = wrap( `I am writing on behalf of my patient, ${name} (Date of Birth: ${formatDate(dob)}), ` + `to request prior authorization for ${requested} for the treatment of ${currentDiagLabel} ` + `(ICD-10: ${currentICD}) affecting ${eyeInBody}.`, y); y+=12; y = wrap( `This patient has demonstrated persistent, recalcitrant disease with inadequate treatment ` + `response despite appropriate trials — defined as three (3) or more injections — of the ` + `following agent(s): ${failedList}. Despite these treatments, the patient has not achieved ` + `adequate anatomic or functional response, as evidenced by persistent subretinal or ` + `intraretinal fluid on optical coherence tomography (OCT) and/or ongoing visual impairment.`, y); y+=12; y = wrap( `Please note that ranibizumab (Lucentis), biosimilar ranibizumab (Byooviz), and ` + `ranibizumab-nuna (Cimerli) are not stocked at our institution and are therefore not ` + `viable alternatives for this patient.`, y); y+=12; y = wrap( `Given the documented treatment failure described above, it is my medical opinion that ` + `${requested} represents the most appropriate and medically necessary next step in the ` + `management of this patient's condition. I respectfully request expedited review and ` + `approval of this prior authorization.`, y); y+=12; y = wrap( `Please do not hesitate to contact our office if additional clinical documentation is required.`, y); y+=32; // Signature sf('normal',10.5,[0,0,0]); doc.text('Sincerely,', mL, y); y+=36; doc.setFont('times','italic'); doc.setFontSize(18); doc.setTextColor(26,46,74); doc.text(signer, mL, y); y+=8; doc.setDrawColor(160,160,160); doc.setLineWidth(0.5); doc.line(mL, y, mL+220, y); y+=14; sf('normal',10.5,[0,0,0]); doc.text(signer, mL, y); y+=13; sf('normal',10,[80,80,80]); doc.text('Department of Ophthalmology — Vitreoretinal Surgery', mL, y); // Footer doc.setDrawColor(200,210,220); doc.setLineWidth(0.5); doc.line(mL,756,pW-mR,756); sf('normal',8,[160,160,160]); doc.text( "This letter was generated for prior authorization purposes. Contact the treating physician's office for questions.", mL, 770 ); const safeName = name.replace(/[^a-zA-Z0-9]/g,'_'); doc.save(`MedNecessity_${safeName}_${new Date().toISOString().slice(0,10)}.pdf`); } function generatePDF() { const name = document.getElementById('patientName').value.trim(); const dob = document.getElementById('dob').value; const requested = document.getElementById('requestedDrug').value; const signer = document.getElementById('signerName').value.trim(); const failed = Array.from(document.querySelectorAll('.checkbox-grid input:checked')).map(c=>c.value); const errEl = document.getElementById('errorMsg'); if (!name || !dob || !currentICD || !requested || !signer || failed.length===0) { errEl.style.display = 'block'; return; } errEl.style.display = 'none'; if (logoDataURL) { buildPDF(logoDataURL); } else { const img = new Image(); img.crossOrigin = 'anonymous'; img.src = 'MSHS-NYEE-Logo-300x12.png'; img.onload = function() { const c = document.createElement('canvas'); c.width = img.naturalWidth; c.height = img.naturalHeight; c.getContext('2d').drawImage(img, 0, 0); buildPDF(c.toDataURL('image/png')); }; img.onerror = () => buildPDF(null); } }
